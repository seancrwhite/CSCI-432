\section*{Question 1: Write one algorithm from EPI in pseudocode.}
  \begin{algorithm}\caption{\textsc{BINARY-SEARCH(X,C)}}\label{alg:bs}
 \begin{algorithmic}[1]
   \State {\bf Input:} $S$, $X,C$
   \State {\bf Output:} $B$, TRUE if value exists in array, otherwise FALSE
   \State ~

   \If{$|X| == 1 && X[0] != C$}
   \State return FALSE
   \EndIf
   
   \State $i \gets |X|/2$

   \If{$X[i] == C$}
   \State return TRUE
   \EndIf

   \If{$X[i] > C$}
   \State return BINARY-SEARCH(X[i:|X|],C)
   \EndIf

   \If{$X[i] > C$}
   \State return BINARY-SEARCH(X[0:i],C)
   \EndIf
 \end{algorithmic}
\end{algorithm}
